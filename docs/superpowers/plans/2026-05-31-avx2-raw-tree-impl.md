# AVX2 + Raw Feature Tree Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Cut remote p99 from 30ms to <1ms by (1) routing ~95% of requests through a raw-feature decision tree before vectorization, and (2) speeding up the remaining IVF k-NN calls with AVX2 SIMD.

**Architecture:** Current pipeline `fast_path → vectorize → decision_tree → IVF` becomes `fast_path → raw_tree(21 raw features) → vectorize → IVF(AVX2)`. Each stage returns `(fraudCount int, hit bool)`; hit=true short-circuits. Index padded from 14 to 16 int16 dims for two clean YMM passes in the SIMD kernel.

**Tech Stack:** Go 1.26 (stdlib + plan9 assembly), Python/uv (index rebuild + ML training), scikit-learn (DecisionTreeClassifier), CGO_ENABLED=0

```python
def build_ivf(vectors, labels, dst):
    n = len(labels)
    # Pad from 14 to 16 dims
    vectors_16 = np.zeros((n, 16), dtype=np.float32)
    vectors_16[:, :14] = vectors
    vectors = vectors_16

    print(f"Fitting MiniBatchKMeans with {N_CLUSTERS} clusters...")
    km = MiniBatchKMeans(
        n_clusters=N_CLUSTERS,
        random_state=42,
        n_init=3,
        batch_size=100_000,
        verbose=0,
    )
    assignments = km.fit_predict(vectors)
    centroids = km.cluster_centers_.astype(np.float32)  # shape: (C, 16)
    print(f"K-means done. Centroids: {centroids.shape}")

    sort_idx = np.argsort(assignments, kind="stable")
    vectors_sorted = vectors[sort_idx]
    labels_sorted = labels[sort_idx]

    cluster_sizes = np.bincount(assignments, minlength=N_CLUSTERS).astype(np.uint32)
    cluster_starts = np.zeros(N_CLUSTERS, dtype=np.uint32)
    cluster_starts[1:] = np.cumsum(cluster_sizes[:-1])

    vectors_int16 = np.clip(np.round(vectors_sorted * 10000), -32768, 32767).astype(
        np.int16
    )

    print("Writing IVF2 index...")
    with open(dst, "wb") as out:
        out.write(b"IVF2")                                  # ← changed
        out.write(struct.pack("<II", N_CLUSTERS, n))
        out.write(centroids.astype("<f4").tobytes())        # 16-dim centroids
        out.write(cluster_starts.astype("<u4").tobytes())
        out.write(cluster_sizes.astype("<u4").tobytes())
        out.write(vectors_int16.astype("<i2").tobytes())    # 16-dim int16 vecs
        out.write(labels_sorted.astype("u1").tobytes())

    size_mb = dst.stat().st_size / 1024 / 1024
    print(f"{n} vectors, {N_CLUSTERS} clusters → {dst} ({size_mb:.1f} MB)")
```

```go
func (idx *IVFIndex) KNN(query [16]float32, k int) int {

	if np > idx.C {
		np = idx.C
	}

	q0 := query[0]
	q1 := query[1]

		d10 := q10 - cents[base+10]
		d11 := q11 - cents[base+11]
		d12 := q12 - cents[base+12]
		d13 := q13 - cents[base+13]

			d7*d7 + d8*d8 + d9*d9 + d10*d10 + d11*d11 + d12*d12 + d13*d13 +
			d14*d14 + d15*d15

		if len(topC) < np {
			topC = append(topC, centEntry{d, c})
			if len(topC) == np {
				maxCD, maxCP = centFindMax(topC)
			}
		} else if d < maxCD {
			topC[maxCP] = centEntry{d, c}

			d7 := q7 - float32(vecs[base+7])*invScale
			dist += d1*d1 + d2*d2 + d3*d3 + d4*d4 + d5*d5 + d6*d6 + d7*d7

			d9 := q9 - float32(vecs[base+9])*invScale
			d10 := q10 - float32(vecs[base+10])*invScale
			d11 := q11 - float32(vecs[base+11])*invScale
			d12 := q12 - float32(vecs[base+12])*invScale

			d14 := q14 - float32(vecs[base+14])*invScale

			dist += d8*d8 + d9*d9 + d10*d10 + d11*d11 + d12*d12 + d13*d13 +

				}
			} else if dist < maxDist {
				top[maxPos] = knnEntry{dist, labs[vi]}
				maxDist, maxPos = knnFindMax(top)
			}
		}
	}

	fraudCount := 0
	for _, e := range top {
		if e.label == 1 {
			fraudCount++
		}
	}
	return fraudCount
}

	maxPos = 0
	for i := 1; i < len(entries); i++ {

			maxDist = entries[i].dist
			maxPos = i
		}
	}

}
```

- [ ] **Step 2.5: Create knn_amd64.s — Plan9 AMD64 AVX2 implementation**

The function uses ABI0 (stack-based). Argument layout:
```
vecs ptr  = 0(FP), 8 bytes
vecs len  = 8(FP), 8 bytes
vecs cap  = 16(FP), 8 bytes
base      = 24(FP), 8 bytes
q         = 32(FP), 8 bytes (pointer to [16]float32)
ret       = 40(FP), 4 bytes (float32)
argsize   = 44 bytes
```

Algorithm: broadcast `1/10000` into Y6; load 8 int16 → sign-extend → cvt to f32 → scale → subtract q → square; repeat for dims 8-15; horizontal sum; return.

```asm
// Copyright: see project license.
#include "textflag.h"

// func distL2i16_16(vecs []int16, base int, q *[16]float32) float32
TEXT ·distL2i16_16(SB), NOSPLIT, $0-44
    MOVQ vecs+0(FP), SI        // SI = &vecs[0]
    MOVQ base+24(FP), AX       // AX = base (element index)
    MOVQ q+32(FP), DI          // DI = q (pointer to [16]float32)

    // Advance SI to &vecs[base]: each int16 is 2 bytes
    LEAQ (SI)(AX*2), SI

    // Y6 = broadcast(float32(1.0/10000.0))
    // float32 bit pattern of 0.0001: verified with math.Float32bits(float32(1.0/10000.0))
    // = 0x38D1B717
    MOVL $0x38D1B717, AX
    VMOVD AX, X6
    VBROADCASTSS X6, Y6

    // Dims 0–7: load 8 × int16 from (SI), sign-extend to int32, convert to f32, scale, diff, square
    VPMOVSXWD (SI), Y0          // Y0 = int32(vecs[base+0..7])
    VCVTDQ2PS Y0, Y0            // Y0 = float32(...)
    VMULPS Y6, Y0, Y0           // Y0 = Y0 * (1/10000)
    VMOVUPS (DI), Y7            // Y7 = q[0..7]
    VSUBPS Y7, Y0, Y0           // Y0 = Y0 - Y7  (diff; squared so sign doesn't matter)
    VMULPS Y0, Y0, Y0           // Y0 = diff²

    // Dims 8–15: SI+16 (8 more int16 = 16 bytes), DI+32 (8 more float32 = 32 bytes)
    VPMOVSXWD 16(SI), Y1        // Y1 = int32(vecs[base+8..15])
    VCVTDQ2PS Y1, Y1
    VMULPS Y6, Y1, Y1
    VMOVUPS 32(DI), Y7          // Y7 = q[8..15]
    VSUBPS Y7, Y1, Y1
    VMULPS Y1, Y1, Y1           // Y1 = diff²

    // Accumulate
    VADDPS Y1, Y0, Y0           // Y0 = sum of all 16 squared diffs

    // Horizontal reduction: 8 floats → 1 float
    VEXTRACTF128 $1, Y0, X1     // X1 = Y0[4..7] (upper lane)
    VADDPS X1, X0, X0           // X0 = X0[0..3] + X1[0..3] (4 floats)
    VHADDPS X0, X0, X0          // X0 = {a0+a1, a2+a3, a0+a1, a2+a3}
    VHADDPS X0, X0, X0          // X0 = {sum, sum, sum, sum}

    VMOVSS X0, ret+40(FP)
    VZEROALL
    RET
```

**Note on VSUBPS direction:** `VSUBPS src1, src2, dst` in Go plan9 means `dst = src2 - src1`. So `VSUBPS Y7, Y0, Y0` → `Y0 = Y0 - Y7` (scaled_vector - query). Since we square the result, the sign is irrelevant. Distance is correct either way.

**Note on VZEROALL:** Required before returning to avoid AVX-SSE transition penalties in calling Go code that uses SSE. This is the standard convention for Go AVX2 assembly.

```python
    out_dir.mkdir(exist_ok=True)
    dst = out_dir / "raw_tree.pkl"

    with open(mcc_path) as f:
        mcc_risk = json.load(f)

    print("Loading records...")
    with gzip.open(src) as f:
        records = json.load(f)
    print(f"Loaded {len(records)} records")

    t0 = time.time()
    X = np.array([extract_features(r, mcc_risk) for r in records], dtype=np.float32)
    y = np.array([1 if r["label"] == "fraud" else 0 for r in records], dtype=np.int8)
    print(f"Feature extraction: {time.time()-t0:.1f}s")

    X_train, X_val, y_train, y_val = train_test_split(
        X, y, test_size=0.1, random_state=42, stratify=y
    )

    print("Training DecisionTreeClassifier...")
    clf = DecisionTreeClassifier(max_depth=None, min_samples_leaf=MIN_SAMPLES_LEAF, random_state=42)
    clf.fit(X_train, y_train)
    n_nodes = clf.tree_.n_node_samples.shape[0]
    print(f"Tree: {n_nodes} nodes")

    # Identify confident leaves
    tree = clf.tree_
    n = tree.node_count
    left = tree.children_left.tolist()
    right = tree.children_right.tolist()
    feature = tree.feature.tolist()
    threshold = tree.threshold.tolist()
    values = tree.value  # shape: (n_nodes, 1, n_classes)

    confident = []
    leaf_class = []
    for i in range(n):
        if left[i] == -1:  # leaf
            counts = values[i, 0]
            total = counts.sum()
            majority_class = int(np.argmax(counts))
            confidence = counts[majority_class] / total if total > 0 else 0
            confident.append(confidence >= CONFIDENCE_THRESHOLD)
            leaf_class.append(majority_class)
        else:
            confident.append(False)
            leaf_class.append(0)

    # Measure coverage on validation set
    covered = sum(1 for xi in X_val if confident[_traverse(xi, left, right, feature, threshold)])
    coverage_pct = covered / len(X_val) * 100

    # FP/FN on covered validation samples
    fp = fn = 0
    for xi, yi in zip(X_val, y_val):
        node = _traverse(xi, left, right, feature, threshold)
        if confident[node]:
            pred = leaf_class[node]
            if pred == 1 and yi == 0:
                fp += 1
            elif pred == 0 and yi == 1:
                fn += 1

    print(f"\nValidation coverage: {coverage_pct:.1f}% of {len(X_val)} samples")
    print(f"FP (confident wrong legit): {fp}, FN (confident wrong fraud): {fn}")
    print(f"Confident leaves: {sum(confident)}")

    import pickle
    model = {
        "n_nodes": n,
        "children_left": left,
        "children_right": right,
        "feature": feature,
        "threshold": [float(x) for x in threshold],
        "confident": confident,
        "leaf_class": leaf_class,
        "feature_names": FEATURE_NAMES,
        "confidence_threshold": CONFIDENCE_THRESHOLD,
    }
    with open(dst, "wb") as f:
        pickle.dump(model, f)
    print(f"\nSaved {dst} ({dst.stat().st_size // 1024} KB)")


def _traverse(x, left, right, feature, threshold):
    node = 0
    while left[node] != -1:
        if x[feature[node]] <= threshold[node]:
            node = left[node]
        else:
            node = right[node]
    return node
```

```go
package service

import (
	"gopher-fraud-detection/internal/dto"

var (

)

// CalculateFraudScore returns fraudCount (0–5): number of fraud neighbors among k=5.
// Pipeline: fast_path → raw_tree → vectorize → IVF k-NN (AVX2).

func CalculateFraudScore(req dto.FraudRequest) int {
	if count, ok := fastPath(req); ok {
		return count
	}
	if count, ok := RawTreePredict(ExtractRawFeatures(req, Vec.MccRisk)); ok {
		return count
	}
	vec := Vec.Vectorize(req)
	return Idx.KNN(vec, 5)
}
```
